
package an.advanced.math.learning.tool;

import java.util.Scanner;

public class AnAdvancedMathLearningTool {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int countCorrect=0;
        int countIncorrect=0;
        int COUNT =0;

        while(COUNT<5){
            int number1 = (int) (Math.random()*10);
            int number2 = (int) (Math.random()*10);
            if(number1<number2){
                int temp = number1;
                number1=number2;
                number2=temp;}
            int result = number1 - number2;
            System.out.println("What is the result of" + number1 + "-" + number2 +"?");
            int userAnswer = input.nextInt();
            if(userAnswer==result){
                System.out.println("Your answer is correct");
                countCorrect++;
            }else{
                System.out.println("Your answer is Incorrect");
                countIncorrect++;
            }
            COUNT++;
                
            }
        System.out.println("number of correct answer is:"+ countCorrect);
        System.out.println("number of Incorrect answer is:"+ countIncorrect);
        }
}
