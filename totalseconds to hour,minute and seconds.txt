package total.seconds.to.hour.minutes.and.seconds;
import java.util.Scanner;

public class TotalSecondsToHourMinutesAndSeconds {
public static void main(String[] args) {
        Scanner input= new Scanner(System.in);
        System.out.println("please enter totalSecondes");
        long totalSeconds = input.nextLong();
        long hours = totalSeconds/3600;
        long reminingSeconds = totalSeconds%3600;
        long minutes = reminingSeconds/60;
        long finalseconds = reminingSeconds%60;
        long seconds =  finalseconds;
        System.out.println(hours +"hours" +","+minutes +"minutes"+  "and"  +seconds+"seconds");
    }
    
}
